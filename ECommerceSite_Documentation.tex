\begin{abstract}
This document provides comprehensive technical documentation for the dappr SQUAD E-Commerce Platform, a full-stack web application designed for premium men's fashion retail and rental services. The platform features a React-based frontend with Vite build system and a Node.js/Express backend API with MongoDB database. The system supports product browsing, user authentication, enquiry management, image management, newsletter subscriptions, and administrative dashboard functionalities. This documentation covers architecture, API specifications, database schemas, deployment procedures, and maintenance guidelines.
\end{abstract}

\section{Introduction}

\subsection{Project Overview}
The dappr SQUAD E-Commerce Platform is a modern, full-stack web application designed to provide premium men's fashion retail and rental services. The platform enables customers to browse, rent, and purchase formal wear including suits, kurtas, bandhgalas, and traditional attire. The system is built with a focus on user experience, performance, security, and SEO optimization.

\subsection{Key Features}
\begin{itemize}
    \item User authentication and authorization (JWT-based)
    \item Product browsing and filtering (Buy/Rent categories)
    \item Enquiry management system
    \item Image management with SEO optimization
    \item Newsletter subscription system
    \item Administrative dashboard
    \item City-based landing pages for local SEO
    \item Responsive mobile-first design
    \item Search engine optimization (SEO) features
    \item WhatsApp integration for customer support
\end{itemize}

\subsection{Technology Stack}
\begin{description}
    \item[Frontend:] React 19.1.1, Vite 7.1.2, React Router DOM 7.8.2, Bootstrap 5.3.8
    \item[Backend:] Node.js, Express.js 5.1.0, MongoDB 8.18.0 (Mongoose)
    \item[Authentication:] JSON Web Tokens (JWT), bcryptjs
    \item[Documentation:] Swagger/OpenAPI
    \item[Deployment:] Vercel (Frontend), Render (Backend)
    \item[Additional:] Axios, React Hook Form, Redux Toolkit, Nodemailer, Twilio
\end{description}

\section{System Architecture}

\subsection{High-Level Architecture}
The system follows a three-tier architecture pattern:

\begin{enumerate}
    \item \textbf{Presentation Layer:} React-based single-page application (SPA) deployed on Vercel
    \item \textbf{Application Layer:} Node.js/Express REST API deployed on Render
    \item \textbf{Data Layer:} MongoDB database hosted on MongoDB Atlas or Render
\end{enumerate}

\section{Database Schema}

\subsection{User Model}
The User model handles user authentication and profile information.

\begin{lstlisting}[caption=User Schema]
{
  fullName: String (required, max: 50),
  email: String (required, unique, lowercase, email format),
  phone: String (required, unique, 10-digit pattern),
  password: String (required, min: 6, hashed with bcrypt),
  isActive: Boolean (default: true),
  resetPasswordToken: String,
  resetPasswordExpires: Date,
  createdAt: Date (auto),
  updatedAt: Date (auto)
}
\end{lstlisting}

\textbf{Indexes:}
\begin{itemize}
    \item Unique index on \texttt{email}
    \item Unique index on \texttt{phone}
\end{itemize}

\textbf{Methods:}
\begin{itemize}
    \item \texttt{comparePassword(candidatePassword)} - Compare password with hash
    \item \texttt{generateResetToken()} - Generate password reset token
    \item \texttt{clearResetToken()} - Clear reset token
\end{itemize}

\subsection{Enquiry Model}
The Enquiry model manages customer enquiries for product rental or purchase.

\begin{lstlisting}[caption=Enquiry Schema]
{
  // User Information
  fullName: String (required, max: 100),
  mobileNumber: String (required, pattern: /^[6-9]\d{9}$/),
  email: String (required, email format),
  
  // Enquiry Type
  enquiryType: String (enum: ['rent', 'buy'], default: 'rent'),
  
  // Booking Information
  preferredBookingDate: Date,
  selectedSize: String,
  selectedQuantity: Number (min: 1, default: 1),
  pickupDate: Date,
  returnDate: Date,
  city: String (required, enum: Kerala districts),
  specialNotes: String (max: 500),
  
  // Product Information
  productId: String,
  productName: String (max: 200),
  
  // Status Management
  status: String (enum: ['new', 'contacted', 'interested', 
                          'converted', 'cancelled'], 
                   default: 'new'),
  priority: String (enum: ['low', 'medium', 'high'], 
                    default: 'medium'),
  adminNotes: String (max: 1000),
  
  // Contact Tracking
  contactedAt: Date,
  contactedBy: String,
  
  // Timestamps
  createdAt: Date (auto),
  updatedAt: Date (auto)
}
\end{lstlisting}

\textbf{Indexes:}
\begin{itemize}
    \item \texttt{\{status: 1, createdAt: -1\}} - Status-based queries
    \item \texttt{\{city: 1\}} - City-based filtering
    \item \texttt{\{preferredBookingDate: 1\}} - Date-based queries
    \item \texttt{\{email: 1\}} - Email lookups
\end{itemize}

\textbf{Virtual Fields:}
\begin{itemize}
    \item \texttt{formattedBookingDate} - Human-readable booking date
    \item \texttt{enquiryAge} - Days since enquiry creation
\end{itemize}

\subsection{Image Model}
The Image model manages product images with SEO optimization features.

\begin{lstlisting}[caption=Image Schema]
{
  // Basic Information
  title: String (required, max: 100),
  description: String (required, max: 500),
  imageUrl: String (required),
  altText: String (required, max: 125),
  
  // SEO Fields
  seoTitle: String (max: 60),
  seoDescription: String (max: 160),
  focusKeyword: String (max: 50, lowercase),
  
  // Categorization
  category: String (enum: ['buy', 'rent', 'featured', 'trending', 
                           'topCategories', 'suits', 'kurtas', 
                           'bandhgalas', 'formal', 'traditional', ...],
                    default: 'rent'),
  categories: [String] (array of categories),
  tags: [String] (lowercase),
  
  // Status
  isActive: Boolean (default: true),
  displayOrder: Number (default: 0),
  
  // Product Information
  price: Number (min: 0),
  rentalPrice: Number (min: 0),
  actualPrice: Number (min: 0),
  securityDeposit: Number (min: 0),
  fabric: String,
  color: String,
  style: String,
  inclusions: String,
  inStock: Boolean (default: true),
  
  // Metadata
  metadata: {
    fileSize: Number (min: 0),
    dimensions: {
      width: Number (min: 1),
      height: Number (min: 1)
    },
    format: String (enum: ['webp', 'jpg', 'jpeg', 'png', 'gif', 'svg']),
    originalFormat: String,
    compressionQuality: Number (min: 1, max: 100, default: 85),
    isOptimized: Boolean (default: false)
  },
  
  uploadedBy: String (required, default: 'admin'),
  createdAt: Date (auto),
  updatedAt: Date (auto)
}
\end{lstlisting}

\textbf{Indexes:}
\begin{itemize}
    \item \texttt{\{category: 1, isActive: 1, displayOrder: 1\}} - Category queries
    \item \texttt{\{tags: 1\}} - Tag-based search
    \item \texttt{\{createdAt: -1\}} - Recent images
\end{itemize}

\textbf{Virtual Fields:}
\begin{itemize}
    \item \texttt{formattedFileSize} - Human-readable file size
    \item \texttt{formattedDimensions} - Formatted dimensions string
    \item \texttt{seoScore} - Calculated SEO score (0-100)
    \item \texttt{seoRecommendations} - Array of SEO improvement suggestions
\end{itemize}

\subsection{Newsletter Model}
The Newsletter model manages email newsletter subscriptions.

\begin{lstlisting}[caption=Newsletter Schema]
{
  email: String (required, unique, lowercase, email format),
  status: String (enum: ['active', 'unsubscribed', 'bounced'], 
                  default: 'active'),
  subscriptionDate: Date (default: Date.now),
  source: String (enum: ['homepage', 'footer', 'popup', 'checkout'],
                  default: 'homepage'),
  preferences: {
    fashionUpdates: Boolean (default: true),
    exclusiveOffers: Boolean (default: true),
    newProducts: Boolean (default: true),
    styleTips: Boolean (default: true)
  },
  metadata: {
    ipAddress: String,
    userAgent: String,
    referrer: String,
    country: String,
    city: String
  },
  createdAt: Date (auto),
  updatedAt: Date (auto)
}
\end{lstlisting}

\textbf{Indexes:}
\begin{itemize}
    \item Unique index on \texttt{email}
    \item \texttt{\{status: 1\}} - Status filtering
    \item \texttt{\{subscriptionDate: -1\}} - Recent subscriptions
\end{itemize}

\textbf{Methods:}
\begin{itemize}
    \item \texttt{getStats()} - Get subscription statistics
    \item \texttt{unsubscribe()} - Mark as unsubscribed
    \item \texttt{resubscribe()} - Reactivate subscription
\end{itemize}

\section{API Documentation}

\subsection{Authentication Endpoints}

\subsubsection{POST /api/auth/register}
Register a new user account.

\textbf{Request Body:}
\begin{lstlisting}[language=json]
{
  "fullName": "John Doe",
  "email": "john@example.com",
  "phone": "9876543210",
  "password": "securepassword123"
}
\end{lstlisting}

\textbf{Response (201 Created):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "User registered successfully",
  "data": {
    "user": {
      "_id": "...",
      "fullName": "John Doe",
      "email": "john@example.com",
      "phone": "9876543210"
    },
    "token": "jwt_token_here"
  }
}
\end{lstlisting}

\textbf{Validation Rules:}
\begin{itemize}
    \item Full name: Required, max 50 characters
    \item Email: Required, valid email format, unique
    \item Phone: Required, exactly 10 digits, unique
    \item Password: Required, minimum 6 characters
\end{itemize}

\subsubsection{POST /api/auth/login}
Authenticate user and receive JWT token.

\textbf{Request Body:}
\begin{lstlisting}[language=json]
{
  "emailOrPhone": "john@example.com",
  "password": "securepassword123"
}
\end{lstlisting}

\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Login successful",
  "data": {
    "user": {...},
    "token": "jwt_token_here"
  }
}
\end{lstlisting}

    \item \texttt{sortBy} - Sort field (default: createdAt)
    \item \texttt{sortOrder} - Sort direction (asc/desc)
\end{itemize}

\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "data": {
    "enquiries": [...],
    "pagination": {
      "currentPage": 1,
      "totalPages": 5,
      "totalItems": 50,
      "itemsPerPage": 10
    }
  }
}
\end{lstlisting}

    \item Connect GitHub repository to Vercel
    \item Configure build settings:
    \begin{itemize}
        \item Framework: Vite
        \item Build command: \texttt{npm run build}
        \item Output directory: \texttt{dist}
    \end{itemize}
    \item Set environment variables:
    \begin{itemize}
        \item \texttt{VITE\_API\_URL} (if using)

\section{Troubleshooting}

\subsection{Common Issues}

\subsubsection{Database Connection Errors}
\begin{itemize}
    \item Verify MongoDB URI in environment variables
    \item Check network access whitelist in MongoDB Atlas
    \item Ensure database server is running
    \item Verify credentials
\end{itemize}

\subsubsection{CORS Errors}
\begin{itemize}
    \item Add frontend URL to backend CORS whitelist
    \item Verify CORS configuration in \texttt{server.js}
    \item Check for trailing slashes in URLs
\end{itemize}

\subsubsection{Authentication Issues}
\begin{itemize}
    \item Verify JWT\_SECRET is set correctly
    \item Check token expiration time
    \item Verify token is being sent in Authorization header
    \item Clear browser localStorage and cookies
\end{itemize}

\subsubsection{Image Upload Issues}
\begin{itemize}
    \item Verify uploads directory exists
    \item Check file size limits
    \item Verify file permissions
    \item Check disk space
\end{itemize}
